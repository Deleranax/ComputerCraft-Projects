-- Simple button ON/OFF example (MonitorAPI)
-- @license MIT
-- @see MonitorAPI
-- @module Simple_Button

os.loadAPI("MonitorAPI")

mon = peripheral.wrap("top")

function b1_command2()
	MonitorAPI.modifyButton(button1, colors.green, colors.white, b1_command1, "ON")
	-- Put some stuff
end

function b1_command1()
	MonitorAPI.modifyButton(button1, colors.red, colors.white, b1_command2, "OFF")
	-- Put some stuff
end

-- @warning Don't call your function in a button declaration, just put the variable of you function (here b1_command)
button1 = MonitorAPI.addButton(4, 4, 4, 3, colors.green, colors.white, b1_command1, "ON")

-- This part can be run in parallel if your program needs to run something else.
while true do
	MonitorAPI.mainLoop(mon)
	sleep(0.1)
end
